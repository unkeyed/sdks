# EnvironmentVariable

## Example Usage

```typescript
import { EnvironmentVariable } from "@unkey/api/models/components";

let value: EnvironmentVariable = {
  key: "DATABASE_URL",
  kind: "writeonly",
  value: "postgresql://user:pass@host:5432/db",
  description: "Primary database connection string",
  createdAt: 1704067200000,
};
```
