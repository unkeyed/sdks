# EnvironmentBuild

Build settings that control how the app is built.
Omitted until the environment has build settings.


## Example Usage

```typescript
import { EnvironmentBuild } from "@unkey/api/models/components";

let value: EnvironmentBuild = {
  dockerfile: "Dockerfile",
  rootDirectory: ".",
  buildCommand: "pnpm --filter api build",
  watchPaths: [
    "src/**",
  ],
  autoDeploy: true,
};
```
